프로세스

1. 사용자가 검색/채팅창에 프롬프트를 입력한다.
2. LLM이 프롬프트를 확인하여 적절한 mcp tool을 검색한다.
3. mcp tool과 사용자의 입력 쿼리를 mcp client로 전달한다.
4. 키워드를 추출하여 mcp tool에 파라미터를 전달한다.
5. mcp tool을 이용하여 RAG에서 문서를 검색한다.
6. 유사도가 높은 문서가 반환된다.
7. mcp tool에 등록한 로직에 따라 문서를 처리한다.
8. 처리한 문서를 LLM에 반환한다.
9. LLM이 문서를 확인하여 문장을 생성하고 사용자에게 답변한다.

`import asyncio
import json
import sys

import feedparser
import httpx

async def main():
    """The main function of the MCP server."""
    reader = asyncio.StreamReader()
    writer = asyncio.StreamWriter(
        transport=None, protocol=None, reader=reader, writer=sys.stdout
    )

    async with httpx.AsyncClient() as client:
        while not reader.at_eof():
            line = await reader.readline()
            if not line:
                break

            request = json.loads(line)
            request_id = request.get("id")

            if "method" not in request or "params" not in request:
                response = {
                    "jsonrpc": "2.0",
                    "error": {
                        "code": -32600,
                        "message": "Invalid Request",
                    },
                    "id": request_id,
                }
                writer.write(json.dumps(response).encode() + b"\n")
                await writer.drain()
                continue

            method = request["method"]
            params = request["params"]

            if method == "get_news_headlines":
                feed_url = params.get(
                    "url", "https://www.yonhapnewstv.co.kr/browse/feed/"
                )
                try:
                    response_rss = await client.get(feed_url)
                    feed = feedparser.parse(response_rss.text)
                    headlines = [entry.title for entry in feed.entries]
                    response = {
                        "jsonrpc": "2.0",
                        "result": {"headlines": headlines},
                        "id": request_id,
                    }
                except Exception as e:
                    response = {
                        "jsonrpc": "2.0",
                        "error": {
                            "code": -32000,
                            "message": f"Failed to fetch or parse RSS feed: {e}",
                        },
                        "id": request_id,
                    }
            else:
                response = {
                    "jsonrpc": "2.0",
                    "error": {
                        "code": -32601,
                        "message": "Method not found",
                    },
                    "id": request_id,
                }

            writer.write(json.dumps(response, ensure_ascii=False).encode() + b"\n")
            await writer.drain()

if __name__ == "__main__":
    asyncio.run(main())`

`# p태그로 문단 추출
def extract_paragraphs(soup):
    paragraphs = []
    for p in soup.find_all('hp:p'):
        # 중첩된 문단은 제외
        if p.find_parent('hp:p') is not None:
            continue

        if p.find('hp:tbl'):
            continue

        texts = [t.get_text(strip=True) for t in p.find_all('hp:t') if t.get_text(strip=True)]
        combined = ''.join(texts).strip()
        if combined:
            paragraphs.append(combined)
    return paragraphs

# ㅁ, ㅇ, *를 추출하여 리스트에 저장
def parse_bullet_list(data_list):
    sections = []
    current_section = None

    for item in data_list:
        item = item.strip()
        if item.startswith('□'):
            if current_section:
                sections.append(current_section)
            current_section = {
                "key_message": item.lstrip('□').strip(),
                "details": [],
                "footnotes": []
            }
        elif item.startswith('ㅇ'):
            if current_section:
                current_section["details"].append(item.lstrip('ㅇ').strip())
        elif item.startswith('*'):
            if current_section:
                current_section["footnotes"].append(item.lstrip('*').strip())

    # 마지막 섹션도 저장
    if current_section:
        sections.append(current_section)

    return {"sections": sections}

# 추출한 문서를 마크다운 형식으로 변환
def to_markdown(parsed_json):
    md = "# 문서 파싱 결과\n\n"
    for sec in parsed_json['sections']:
        md += f"## {sec['key_message']}\n\n"
        for detail in sec['details']:
            md += f"- {detail}\n"
        if sec['footnotes']:
            md += "\n**참고:**\n"
            for foot in sec['footnotes']:
                md += f"  - {foot}\n"
        md += "\n"
    return md

# tbl로 테이블 추출, 마크다운 형식으로 변환
def extract_markdown_tables(soup):
    markdown_tables = []
    for table in soup.find_all('hp:tbl'):
        rows = []

        for tr in table.find_all('hp:tr'):
            row = []
            for tc in tr.find_all('hp:tc'):
                # 각 셀의 텍스트 수집
                texts = [t.get_text(strip=True) for t in tc.find_all('hp:t')]
                cell_text = ' '.join(texts).strip()
                row.append(cell_text)
            if row:
                rows.append(row)

        # Markdown 변환
        if not rows:
            continue

        # 헤더 처리 (첫 행을 헤더로 간주)
        header = '| ' + ' | '.join(rows[0]) + ' |'
        separator = '| ' + ' | '.join(['---'] * len(rows[0])) + ' |'
        body = ['| ' + ' | '.join(r) + ' |' for r in rows[1:]]

        markdown = '\n'.join([header, separator] + body)
        markdown_tables.append(markdown)

    return markdown_tables`
